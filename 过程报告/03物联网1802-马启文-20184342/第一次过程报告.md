void read_data(const string& file, string& Data)
{
	ifstream fin(file.c_str());
	if (!fin)
	{
		cerr << file << " error!" << endl;
		exit(2);
	}
	Data.clear();
	string line;
	while (getline(fin, line))
	{
		Data += line + '\n';
	}
}
